def get_ans(n, p, s):
    if n == 1:
        return [p[0]]
    
    liked = [i for i in range(n) if s[i] == '1']
    disliked = [i for i in range(n) if s[i] == '0']
    
    res = liked + disliked
    for i in range(n):
        res[i] = p[i]
    
    return res
