# z-function
s = input()
n = len(s)

z = [0] * (n + 1)
for i in range(1, n + 1):
    j = i - 1
    while j >= 0 and s[j] == s[i - j - 1]:
        j -= 1
    z[i] = j + 1


def calc(l, r):
    return sum(z[l:r + 1])


q = int(input())
for _ in range(q):
    l, r = map(int, input().split())
    print(calc(l, r))
